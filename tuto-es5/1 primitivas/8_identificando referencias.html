identificando referencias

  • se usa el operador typeof para identificar funciones,
  • pero para los demas objectos resulta un poco confuso por que siempre regresa "object",
  • por este motivo usamos instanceof
    • function reflect(value){
    return value;
    }
    console.log('que regresa el typeof de una funcion?" typeof reflect -> ' + (typeof reflect)); // function
    var items = [];
    var object = {};
    console.log("items instanceof Array ->" + (items instanceof Array)); // true
    console.log("object instanceof Object ->" + (object instanceof Object)); // true
    console.log("reflect instanceof Function ->" + (reflect instanceof Function)); // true
  • el operador instanceof tambien identifica inherited types
    • function reflect(value){
    return value;
    }
    var items = [];
    var object = {};
    console.log("items instanceof Array ->" + (items instanceof Array)); // true
    console.log("items instanceof Object ->" + (items instanceof Object)); // true
    console.log("object instanceof Object ->" + (object instanceof Object)); // true
    console.log("object instanceof Array ->" + (object instanceof Array)); // false
    console.log("reflect instanceof Function ->" + (reflect instanceof Function)); // true
    console.log("reflect instanceof Object ->" + (reflect instanceof Object)); // true
